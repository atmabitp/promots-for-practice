CREATE OR REPLACE TABLE bfl_std_lake.temp_analytics.april_agreement_dispo_summary AS
SELECT *,
CASE WHEN first_dispose IN (
  'ANSWERED',
  'PTPCB',
  'Want Settlement',
  'RTP (refuse to Pay)',
  'Call Back',
  'Customer Disconnected',
  'Call Disconnected',
  'CLPD (Claims Paid)',
  'Escalation',
  'Death',
  'Foreclosure',
  'Normal Closure'
)THEN 1 else 0 END as first_dispose_n,

case when first_dispose IN ('PTPCB', 'Want Settlement') THEN 1 else 0 end AS ptp_flag,
CASE WHEN first_dispose IN ('CLPD (Claims Paid)', 'Paid_case') THEN 1 else 0 END AS already_paid_calls,

CASE WHEN first_dispose = 'RTP (refuse to Pay)' THEN 1 else 0 END AS rtp_flag,
CASE WHEN first_dispose = 'Want Settlement' THEN 1 else 0 END AS settlement_flag,
CASE WHEN first_dispose = 'Call Back' THEN 1 ELSE 0 END AS callback_flag,

CASE WHEN customer_behaviour = 'Polite/Cooperative' THEN 1 else 0 end as Polite_flag ,
CASE WHEN customer_behaviour = 'Rude/Aggressive' THEN 1 else 0 END as Rude_flag,

case 
when Hour BETWEEN 7 and 10 then  'A_7_11'  --4hr
when Hour BETWEEN 11 and 14 then 'B_11_15' --3hr
when Hour BETWEEN 15 and 16 then 'C_14_17' --
when Hour BETWEEN 17 and 18 then 'D_17_19' 
when Hour >= 19 then 'E_>19' else null end as Time_duration,

row_number() OVER ( PARTITION BY AGREEMENT_NO order by day asc) as First_call,
row_number() OVER ( PARTITION BY AGREEMENT_NO order by day desc) as Last_call


from bfl_std_lake.temp_analytics.temp_april_calls_ab_1

WITH day_list AS (
  SELECT 4 AS Day UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7
  UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10
),
daily_calls AS (
  SELECT
    Day,
    COUNT(*) AS total_calls,
    SUM(CASE WHEN call_answered = 'Y' THEN 1 ELSE 0 END) AS answered_calls,
    COUNT(DISTINCT CASE WHEN call_answered = 'Y' THEN AGREEMENT_NO END) AS unique_loans_answered,

WITH loan_level AS (
  SELECT
    DATE_FORMAT(TO_DATE(CONCAT('2025-04-', LPAD(Day,2,'0')), 'yyyy-MM-dd'), 'EEEE') AS weekday,
    Time_duration,
    AGREEMENT_NO,
    MAX(CASE WHEN call_answered = 'Y' THEN 1 ELSE 0 END) AS answered,
    MAX(same_day_paid) AS paid_same,
    MAX(next_day_paid) AS paid_next
  FROM bfl_std_lake.temp_analytics.april_agreement_dispo_summary_final_ap
  WHERE Day BETWEEN 4 AND 10
    AND TC_ALLC = 1
  GROUP BY weekday, Time_duration, AGREEMENT_NO
)
SELECT
  weekday,
  Time_duration,
  COUNT_IF(answered = 1) AS total_loans_answered,
  SUM(paid_same) AS total_loans_paid_same_day,
  SUM(paid_next) AS total_loans_paid_next_day
FROM loan_level
WHERE answered = 1
GROUP BY weekday, Time_duration
ORDER BY
  CASE weekday WHEN 'Monday' THEN 1 WHEN 'Tuesday' THEN 2 WHEN 'Wednesday' THEN 3
               WHEN 'Thursday' THEN 4 WHEN 'Friday' THEN 5 WHEN 'Saturday' THEN 6
               WHEN 'Sunday' THEN 7 ELSE 8 END,
  Time_duration;
